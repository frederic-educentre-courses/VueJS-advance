Vue.JS avancé

Récupérer de la données d'une API

import { ref, onMounted } from 'vue'

export default {
  setup() {
    const products = ref([]) // tableau réactif pour stocker les produits

    onMounted(async () => { // hook de cycle de vie appelé après le montage du composant
      const response = await fetch('https://api.example.com/products')
      products.value = await response.json()
    })

    return { products } // expose la variable products au template
  }
}

v-for avec fetch

<template>
  <div>
	<ul>
	  <li v-for="product in products" :key="product.id">
		{{ product.name }}
	  </li>
	</ul>
  </div>
</template>

v-if & v-else

<template>
  <div>
	<div v-if="products.length > 0">
	  <ul>
		<li v-for="product in products" :key="product.id">
		  {{ product.name }}
		</li>
	  </ul>
	</div>
	<div v-else>
	  Chargement des produits...
	</div>
  </div>
</template>
